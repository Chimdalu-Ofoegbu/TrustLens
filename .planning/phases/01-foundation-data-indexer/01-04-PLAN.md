    ```python
    """Refresh pipeline: census CSV -> SQLite. Entry point: python -m indexer.refresh"""

    DEFAULT_CSV = Path("data/okx-marketplace-census-2026-07-10.csv")
    DEFAULT_DB = Path("data/trustlens.db")

    @dataclass(frozen=True)
    class RefreshSummary:
        agents: int
        snapshots_appended: int
        field_warnings: int
        source: str

    def persist(conn, records, captured_at: str, source: str = "census") -> None:
        """THE Phase 5 scraper seam: any loader producing AgentRecords calls this
        same path with its own source tag. Caller manages the transaction."""
        for rec in records:
            upsert_agent(conn, rec, captured_at)
            insert_snapshot(conn, rec, captured_at, source)

    def refresh(csv_path, db_path, captured_at: str, source: str = "census") -> RefreshSummary:
        records, warnings = load_census(csv_path)
        conn = connect(db_path)
        try:
            init_db(conn)
            with conn:                    # one atomic transaction for all 544 writes
                persist(conn, records, captured_at, source)
        finally:
            conn.close()                  # clean close also removes WAL sidecars
        return RefreshSummary(len(records), len(records), warnings, source)

    def main(argv: list[str] | None = None) -> int: ...

    if __name__ == "__main__":
        raise SystemExit(main())
    ```
